var env = new ConsoleEnvironment();
ITransformer model;
using (var f = new FileStream("model.zip", FileMode.Open))
model = TransformerChain.LoadFrom(env, f);

var predictor = model.MakePredictionFunction<Feelings, FeelingsType>(env);

Console.Write("\nHow are you feeling? ");
var text = Console.ReadLine().Trim();

while (!string.IsNullOrWhiteSpace(text))
{
var o = predictor.Predict(new Feelings { Text = text });
Console.WriteLine($"\n-------\nText: {text}\nPredicted: {o.Predicted}\n-------\n");
Console.Write("How are you feeling? ");
text = Console.ReadLine().Trim();
}

Console.Write("\nthanks for playing!");
